## Animation Specs

### Hero Word-by-Word Animation
```javascript
headline: {
  words: ['We', 'Build', 'AI-Powered', 'Products', 'That', 'Generate', 'Real Revenue'],
  animation: {
    opacity: [0,1],
    y: [20,0],
    stagger: 0.1,
    duration: 500
  },
  highlightWords: ['Real Revenue'],
  highlightStyle: { 
    background: 'linear-gradient(135deg, #667eea, #4caf50)',
    WebkitBackgroundClip: 'text',
    color: 'transparent'
  }
}

backgroundMesh: {
  gradient: ['#667eea', '#4caf50', '#667eea'],
  animate: { 
    morphing: true,
    duration: 8000,
    loop: true
  }
}

floatingElements: {
  parallax: true,
  scrollMultiplier: 0.3,
  drift: { y: [0,20,0], duration: 4000, loop: true }
}
```

### Service Cards Stagger + Hover
```javascript
cards: {
  initial: { opacity: 0, y: 30 },
  onScrollIntoView: {
    opacity: 1,
    y: 0,
    stagger: 0.15,

  }

card.onHover: {
  y: -8,
  boxShadow: '0 20px 40px rgba(102,126,234,0.3)',
  icon: { rotate: 5 },
  border: {
    background: 'linear-gradient(135deg, #667eea, #4caf50)',
    width: 2
  },
  duration: 300
}
```

### Timeline Draw Animation
```javascript
timeline: {
  trigger: 'scroll into view',
  connector: {
    strokeDasharray: 1000,
    strokeDashoffset: [1000, 0],
    duration: 2000
  },
  phases: {
    stagger: 0.2,
    each: {
      node: { scale: [0,1.2,1], opacity: [0,1] },
      text: { opacity: [0,1], y: [10,0], delay: 0.2 }
    }
  },
  activePhase: {
    scale: [1,1.15,1],
    boxShadow: 'pulse',
    loop: true,
    duration: 2000
  }
}
```
